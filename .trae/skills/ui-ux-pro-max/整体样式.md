// src/components/ThemeProvider.tsx
"use client";

import { createContext, useContext, 
useState, ReactNode, useEffect } from 
"react";
import { themes } from "@/styles/themes";

type ThemeKey = keyof typeof themes;

interface ThemeContextType {
  theme: ThemeKey;
  setTheme: (theme: ThemeKey) => void;
  themeConfig: typeof themes[ThemeKey];
}

const ThemeContext = 
createContext<ThemeContextType | 
undefined>(undefined);

export function ThemeProvider({ children }
: { children: ReactNode }) {
  const [theme, setTheme] = 
  useState<ThemeKey>(
    typeof window !== "undefined"
      ? (localStorage.getItem("theme") as 
      ThemeKey) || "chinese"
      : "chinese"
  );

  // 从服务器获取主题配置
  useEffect(() => {
    const fetchTheme = async () => {
      try {
        const response = await fetch("/
        api/themes");
        if (response.ok) {
          const data = await response.json
          ();
          if (data.theme && themes[data.
          theme]) {
            setTheme(data.theme);
          }
        }
      } catch (error) {
        console.error("Failed to fetch 
        theme:", error);
      }
    };

    fetchTheme();
  }, []);

  // 当主题变化时，更新 localStorage 和 CSS 
  变量
  useEffect(() => {
    if (typeof window !== "undefined") {
      localStorage.setItem("theme", 
      theme);
      updateCSSVariables();
    }
  }, [theme]);

  // 更新 CSS 变量
  const updateCSSVariables = () => {
    const themeConfig = themes[theme];
    const root = document.documentElement;

    root.style.setProperty("--primary", 
    themeConfig.colors.primary);
    root.style.setProperty("--secondary", 
    themeConfig.colors.secondary);
    root.style.setProperty("--accent", 
    themeConfig.colors.accent);
    root.style.setProperty("--dark", 
    themeConfig.colors.dark);
    root.style.setProperty("--light", 
    themeConfig.colors.light);
    root.style.setProperty
    ("--background", themeConfig.colors.
    background);
    root.style.setProperty
    ("--background-alt", themeConfig.
    colors.backgroundAlt);
    root.style.setProperty("--text", 
    themeConfig.colors.text);
    root.style.setProperty
    ("--text-muted", themeConfig.colors.
    textMuted);
    root.style.setProperty("--border", 
    themeConfig.colors.border);
    root.style.setProperty("--card", 
    themeConfig.colors.card);
    root.style.setProperty
    ("--heading-font", themeConfig.fonts.
    heading);
    root.style.setProperty("--body-font", 
    themeConfig.fonts.body);
  };

  return (
    <ThemeContext.Provider value={{ 
    theme, setTheme, themeConfig: themes
    [theme] }}>
      <div className={`theme-${theme}`}>
        {children}
      </div>
    </ThemeContext.Provider>
  );
}

export function useTheme() {
  const context = useContext
  (ThemeContext);
  if (context === undefined) {
    throw new Error("useTheme must be 
    used within a ThemeProvider");
  }
  return context;
}

// src/app/api/themes/route.ts
import { NextRequest, NextResponse } from 
"next/server";
import { db } from "@/lib/db";

export async function GET() {
  try {
    // 从数据库获取主题配置
    const result = await db.get(
      "SELECT value FROM system_configs 
      WHERE key = 'theme'"
    );
    
    return NextResponse.json({
      theme: result?.value || "chinese",
    });
  } catch (error) {
    console.error("Failed to get theme:", 
    error);
    return NextResponse.json({ theme: 
    "chinese" }, { status: 500 });
  }
}

export async function POST(request: 
NextRequest) {
  try {
    const { theme } = await request.json
    ();
    
    // 验证主题是否有效
    const validThemes = ["chinese", 
    "middleEastern", "amazon", 
    "middleEasternLuxury"];
    if (!validThemes.includes(theme)) {
      return NextResponse.json({ error: "
      无效的主题" }, { status: 400 });
    }
    
    // 保存主题配置到数据库
    await db.run(
      "INSERT OR REPLACE INTO 
      system_configs (key, value) VALUES 
      (?, ?)",
      "theme",
      theme
    );
    
    return NextResponse.json({ success: 
    true });
  } catch (error) {
    console.error("Failed to save 
    theme:", error);
    return NextResponse.json({ error: "保
    存失败" }, { status: 500 });
  }
}
